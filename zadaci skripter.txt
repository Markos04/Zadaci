
2)  Napiši program koji određuje broj kućice u kojoj živi Viktor 
(ako je između Ane i Borisa paran broj kućica, Viktor živi u onoj koja je bliža Ani).
(a - b)/2 - razdaljina tacke od centra izmedju tacaka
(a + b)/2 - pozicija centra izmedju tacaka

#include <iostream>

using namespace std;

int main(){
    int br_p, prvi, drugi, do_dr;

    cin >> br_p >> prvi >> drugi;
    if (drugi <= br_p){
        do_dr = drugi - prvi;
    }
    if (drugi > br_p){
        do_dr = (br_p - prvi) + drugi;
    }
    cout << do_dr << endl;

    return 0;
}
